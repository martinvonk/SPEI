                "deficit = evap - prec\n",
                "## deficit period\n",
                "startdate = pd.Timestamp(\"2000-04-01\")\n",
                "enddate = pd.Timestamp(\"2000-09-30\")\n",

                "    deficit=deficit,\n",
                "    startdate=startdate,\n",
                "    enddate=enddate,\n",
                "    allow_below_zero=False,\n",
                ")\n",
                "# plot deficit\n",
                "ax = cumdf.plot(figsize=(7.0, 6.0), cmap=\"cividis\")\n",
